	desc = "EVTDESCancrel.0183"
	picture = "GFX_evt_guardian"
	
	is_triggered_only = yes
	
	option = {
		name = "EVTOPTAancrel.0183" # Send him on the path of the warrior
		ai_chance = {
			factor = 100
			modifier = {
				factor = 0
				FROM = {
					is_female = yes
				}
				has_character_modifier = female_marshals_forbidden
			}
		}
		FROM = { prestige = 20}
		FROM = { change_martial = 1}
		FROM = { change_learning = -1}
	}

	option = {
		name = "EVTOPTBancrel.0183" # Send him to the druids
		ai_chance = {
			factor = 100
		}
		trigger = {
			OR = {
				trait = zealous
				trait = diligent
				trait = patient
				trait = ambitious
				trait = scholar
				trait = poet
				trait = mystic
				trait = druid
				trait = content
			}
			OR = {
				FROM = { NOT = { is_ruler = yes }}
				FROM = { is_theocracy = yes }
			}
		}
		custom_tooltip = {
			text = path_of_the_druids
			FROM = {piety = 50}
			FROM = { change_learning = 1}
			FROM = { change_martial = -1}
			hidden_tooltip = {
				FROM = {
					narrative_event = {
						id = ancrel.0185
						days = 3
					}
				}
			}
		}
	}
}

# Child wants to become a druid (Child)
character_event = {
	id = ancrel.0186
	hide_window = yes
	
	ai = yes
	min_age = 16
	max_age = 25
	only_men = yes
	capable_only = yes
	prisoner = no
	
	trigger = {
		has_dlc = "The Old Gods"
		is_landed = no
		in_command = no
		religion_group = celtic_group
		NOT = {
			OR = {
				trait = cynical
				trait = imbecile
				trait = slow
				trait = craven
			}
			has_character_flag = wants_druidic_training
		}
		OR = {
			father = {
				is_alive = yes
				religion_group = celtic_group
				is_ruler = yes
				is_liege_of = ROOT
				prisoner = no
				NOT = { trait = incapable }
				ROOT = { host = { character = PREVPREV } }
			}
			mother = {
				is_alive = yes
				religion_group = celtic_group
				is_ruler = yes
				is_liege_of = ROOT
				prisoner = no
				NOT = { trait = incapable }
				ROOT = { host = { character = PREVPREV } }
			}
		}		
	}
	
	mean_time_to_happen = {
		months = 450
		
		modifier = {
			factor = 0.5
			trait = content
		}
		modifier = {
			factor = 0.5
			trait = brave
		}
		modifier = {
			factor = 0.75
			trait = quick
		}
		modifier = {
			factor = 0.75
			trait = genius
		}
		modifier = {
			factor = 1.5
			trait = cynical
		}
		modifier = {
			factor = 1.5
			trait = ambitious
		}
		modifier = {
			factor = 2.0
			is_primary_heir = yes
		}
		modifier = {
			factor = 2.0
			trait = hunter
		}
	}
	
	immediate = {
		if = {
			limit = {
				father = {
					is_alive = yes
					ROOT = {
						liege = {
							character = PREVPREV
						}
					}
				}
			}
			father = { character_event = { id = ancrel.0187 } }
		}
		if = {
			limit = {
				mother = {
					is_alive = yes
					ROOT = {
						liege = {
							character = PREVPREV
						}
					}
				}
			}
			mother = { character_event = { id = ancrel.0187 } }
		}
		set_character_flag = wants_druidic_training
	}
	
	option = {
		name = OK
	}
}
