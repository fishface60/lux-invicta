# Son wants to join the Akademia (Son)
character_event = {
	id = academia.1
	hide_window = yes
	
	ai = yes
	min_age = 16
	max_age = 40
	only_men = yes
	capable_only = yes
	prisoner = no
	
	trigger = {
		is_landed = no
		in_command = no
		OR = {
			religion_group = neoplato_group
			religion = valentinian
			religion = docetist
			religion = hurufi
			religion = ilm_islam
			religion = marcionist
		}
		
		NOT = { 
			trait = scholarlystudent 
			has_character_flag = wants_scholar
		}
		father = {
			is_alive = yes
			OR = {
				religion_group = neoplato_group
				religion = valentinian
				religion = docetist
				religion = hurufi
				religion = ilm_islam
				religion = marcionist
			}
			is_ruler = yes
			is_liege_of = ROOT
			prisoner = no
			NOT = { trait = incapable }
			ROOT = { host = { character = PREVPREV } }
		}
	}
	
	weight_multiplier = {
		months = 100
		
		modifier = {
			factor = 0.2
			trait = genius
		}
		
		modifier = {
			factor = 0.5
			trait = quick
		}
		
		modifier = {
			factor = 0.5
			trait = ambitious
		}
		modifier = {
			factor = 0.5
			trait = philosophical
		}
		modifier = {
			factor = 0.75
			trait = scholar
		}
		modifier = {
			factor = 1.5
			trait = slothful
		}
		modifier = {
			factor = 1.5
			trait = content
		}
		modifier = {
			factor = 1.5
			is_primary_heir = yes
		}
		modifier = {
			factor = 1.5
			trait = imbecile
		}
		modifier = {
			factor = 2.0
			trait = slow
		}
	}
	
	immediate = {
		father = { character_event = { id = academia.2 } }
		set_character_flag = wants_scholar
	}
	
	option = {
		name = OK
	}
}
