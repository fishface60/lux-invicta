character_event = {
	id = lux_nm_court.1
	desc = "EVTDESClux_nm_court.1"
	picture = GFX_evt_monk
	
	only_rulers = yes
	
	trigger = {
		OR = {
			religion = norse_pagan
			religion = ragnarok
			religion = norse_war
			religion = sowilo
		}
		any_demesne_title = {
			is_theocracy = yes
			holder_scope = {
				is_female = no
				OR = {
					religion = norse_pagan
					religion = ragnarok
					religion = norse_war
					religion = sowilo
				}
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = EVTOPTAlux_nm_court.1
		random_demesne_title = {
			limit = {
				is_theocracy = yes
			}
			if = {
				limit = {
					holder_scope = {
						NOT = {
							any_courtier = {
								prisoner = no
								NOT = {
									trait = incapable
								}
								is_female = yes
								age = 16
								is_ruler = no
							}
						}
					}
				}
				create_random_priest = {
					random_traits = yes
					age = 16
					dynasty = random
					female = yes
					culture = PREV
				}
			}
			holder_scope = {
				random_courtier = {
					limit = {
						prisoner = no
						NOT = {
							trait = incapable
						}
						is_female = yes
						age = 16
						is_ruler = no
					}
					grant_title = PREVPREV
				}
			}
		}
	}
}

	id = lux_nm_court.2
	desc = "EVTDESClux_nm_court.2"
	picture = GFX_evt_spymaster
	
	only_men = yes
	min_age = 16
	capable_only = yes
	prisoner = no
	
	trigger = {
		is_councillor = no
		independent = no
		NOT = {
			liege = {
				has_character_flag = asked_marshal_petition
			}
		}
		NOT = {
			has_character_flag = asked_for_job
		}
		NOT = {
			liege = {
				any_courtier = {
					has_job_title = job_marshal
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 12
		modifier = {
			factor = 0.25
			trait = ambitious
		}
		modifier = {
			factor = 2.0
			trait = content
		}
		modifier = {
			factor = 4.0
			trait = slothful
		}
		modifier = {
			factor = 2.5
			trait = procrastinator
		}
		modifier = {
			factor = 2.0
			trait = humble
		}
		modifier = {
			factor = 0.5
			trait = proud
		}
		modifier = {
			factor = 1.1
			NOT = {
				martial = 5
			}
		}
		modifier = {
			factor = 0.9
			martial = 13
		}
		modifier = {
			factor = 0.75
			martial = 20
		}
		modifier = {
			factor = 0.1
			trait = lunatic
		}
	}
	
	immediate = {
		set_character_flag = asked_for_job
		liege = {
			set_character_flag = asked_marshal_petition
		}
	}
	
	option = {
		name = "EVTOPTAlux_nm_court.2"
		liege = {
			letter_event = {
				id = lux_nm_court.3
				tooltip = EVTTOOLTIPlux_nm_court.3
			}
		}
	}
}
