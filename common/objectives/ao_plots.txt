# Make liege your lover
plot_seduce_liege = {
	type = realm_characters
	intrigue_plot = yes
	
	# Plotter scope
	potential = {
		prisoner = no
		age = 16
		NOT = { age = 45 }
		is_female = yes
		has_lover = no
		OR={
			is_consort = no
			ROOT={any_consort={character=FROM}}
		}
		NOT = { trait = incapable }
		NOT = { trait = imbecile }
		NOT = { trait = clubfooted }
		NOT = { trait = hunchback }
		NOT = { trait = ugly }
		NOT = { trait = lisp }
		NOT = { trait = inbred }
		NOT = { trait = dwarf }
		NOT = { trait = homosexual }
		NOT = { trait = chaste }
		NOT = { trait = celibate }
		NOT = { health_traits = 1 }
	}
	
	# Target allow trigger for when players target a specific character in the GUI
	player_allow = {
		age = 16
		is_female = no
		has_lover = no
		NOT = { is_close_relative = FROM }
		#NOT = { spouse = { character = FROM } }
		NOT = { trait = celibate }
		NOT = { trait = homosexual }
		NOT = { trait = eunuch }
		FROM = { 
			NOT = { has_opinion_modifier = { who = ROOT modifier = love_scorned } }
		}
	}
	
	# Target character scope
	allow = {
		# Must be male liege with no lover already
		age = 16
		is_female = no
		has_lover = no
		NOT = { is_close_relative = FROM }
		#NOT = { spouse = { character = FROM } }
		NOT = { trait = celibate }
		NOT = { trait = homosexual }
		NOT = { trait = eunuch }
		FROM = { 
			NOT = { has_opinion_modifier = { who = ROOT modifier = love_scorned } }
		}
		OR = {
			is_liege_of = FROM
			father = {
				is_liege_of = FROM
			}
		}
	}
	
	chance = {
		factor = 2
		
		modifier = {
			factor = 0.01
			FROM = { trait = content }
		}
		
		modifier = {
			factor = 2.0
			opinion = { who = FROM value = 40 } 
		}
		
		modifier = {
			factor = 2.0
			FROM = { trait = deceitful }
		}
		
		modifier = {
			factor = 2.5
			FROM = { trait = ambitious }
		}
		
		modifier = {
			factor = 5.0
			FROM = { 
				OR = {
					trait = lustful 
					trait = hedonist
				}
			}
		}
		
		modifier = {
			factor = 1.5
			FROM = { trait = genius }
		}
		modifier = {
			factor = 1.25
			FROM = { trait = gregarious }
		}
		modifier = {
			factor = 1.25
			FROM = { trait = fair }
		}
		modifier = {
			factor = 1.25
			FROM = { trait = strong }
		}
		modifier = {
			factor = 1.5
			FROM = { trait = elusive_shadow }
		}
		
		modifier = {
			factor = 0.01
			trait = chaste
		}
		
		modifier = {
			factor = 1.25
			tier = duke
		}
		
		modifier = {
			factor = 1.5
			OR = {
				tier = king
				tier = emperor
			}
		}
		
		modifier = {
			factor = 0.5
			FROM = { age = 25 }
		}
		
		modifier = {
			factor = 0.05
			FROM = { age = 35 }
		}
		
		modifier = {
			factor = 2.5
			FROM = {
				NOT = { num_of_children = 1 }
			}
		}
		
		modifier = {
			factor = 0.4
			FROM = { num_of_children = 2 }
		}
		
		modifier = {
			factor = 0.2
			FROM = { num_of_children = 3 }
		}
		
		modifier = {
			factor = 0.2
			FROM = { 
				any_child = {
					trait = bastard
				}
			}
		}
		
		modifier = {
			factor = 0.5
			FROM = { is_married = yes }
		}
		
		modifier = {
			factor = 0.05
			FROM = { is_ruler = yes }
		}
		
		modifier = {
			factor = 0.15
			FROM = { 
				spouse = {
					is_ruler = yes
				}
			}
		}
		
	}
	
	success = {
		has_opinion_modifier = { who = FROM modifier = opinion_lover } 
	}
	abort = {
		OR = {
			is_alive = no
			has_lover = yes
			trait = celibate
			FROM = { trait = celibate }
			FROM = { has_lover = yes }
			FROM = { age = 45 }
			FROM = { has_character_flag = plot_seduce_failed }
			FROM = { health_traits = 1 }
		}
	}
	abort_effect = {
		FROM = {
			# Clear flags
			hidden_tooltip = {
				clr_character_flag = plot_seduce_liege_decision_100_taken
				clr_character_flag = seduce_in_motion
				clr_character_flag = plot_seduce_failed
			}
			# Prevents character from targeting same target again too quickly
			opinion = {
				modifier = love_scorned
				who = ROOT
				months = 12
			}
		}
	}
	effect = {
		FROM = {
			prestige = 25
			# Clear flags
			hidden_tooltip = {
				clr_character_flag = plot_seduce_liege_decision_100_taken
				clr_character_flag = seduce_in_motion
				clr_character_flag = plot_seduce_failed
				clr_character_flag = plot_seducer 
			}
			
		}
	}
}

# Man seduces woman
plot_seduce_for_men = {
	type = realm_characters
	intrigue_plot = yes
	
	# Plotter scope
	potential = {
		prisoner = no
		age = 16
		NOT = { age = 55 }
		is_female = no
		NOT={num_of_lovers = 5}
		NOT = { trait = incapable }
		NOT = { trait = imbecile }
		NOT = { trait = clubfooted }
		NOT = { trait = hunchback }
		NOT = { trait = harelip }
		NOT = { trait = ugly }
		NOT = { trait = lisp }
		NOT = { trait = inbred }
		NOT = { trait = dwarf }
		NOT = { trait = homosexual }
		NOT = { trait = chaste }
		NOT = { trait = celibate }
		NOT = { trait = eunuch }
		OR = {
			NOT = { health_traits = 1 }
			trait = lunatic
			trait = possessed
		}
		NOT = { health_traits = 2 }
		
	}
	
	# Target allow trigger for when players target a specific character in the GUI
	player_allow = {
		is_female = yes
		age = 16
		NOT = { age = 45 }
		NOT = { trait = celibate }
		NOT = { trait = homosexual }
		NOT = { is_close_relative = FROM }
		#NOT = { spouse = { character = FROM } }
		is_within_diplo_range = FROM
		NOT={num_of_lovers = 5}
		FROM = { 
			NOT = { has_opinion_modifier = { who = ROOT modifier = love_scorned } }
		}
	}
	
	# Target character scope
	allow = {
		# Some basic rules regardless of target
		NOT = { is_close_relative = FROM }
		NOT = { spouse = { character = FROM } }
		age = 16
		NOT = { age = 45 }
		NOT = { trait = celibate }
		NOT = { trait = homosexual }
		is_within_diplo_range = FROM
		is_female = yes
		NOT={num_of_lovers = 5}
		FROM = { 
			NOT = { has_opinion_modifier = { who = ROOT modifier = love_scorned } }
		}
	
		OR = {
			# female liege
				is_liege_of = FROM
		
			# landed female vassal
			AND = {
				FROM = {
					is_liege_of = ROOT
				}
				is_ruler = yes
			}
			
			# female courtier with claim, or heir or pretender
			AND = {
				FROM = {
					is_liege_of = ROOT
				}
				is_ruler = no
				OR = {
					num_of_claims = 1
					any_pretender_title = {always = yes}
					any_heir_title = {always = yes}
				}
			}
			
			# crazy and wants to sleep with vassal's wife
			FROM = {
				OR = { trait = lunatic trait = possessed }
				any_vassal = { spouse = { character = ROOT }}
			}
		}
	}
	
	chance = {
		factor = 2
		
		modifier = {
			factor = 0.01
			FROM = { trait = content }
		}
		
		modifier = {
			factor = 2.0
			opinion = { who = FROM value = 40 } 
		}
		
		modifier = {
			factor = 0.25
			is_liege_of = FROM
		}
		
		modifier = {
			factor = 2.0
			FROM = { trait = arbitrary } 
		}
		
		modifier = {
			factor = 2.0
			FROM = { trait = deceitful }
		}
		
		modifier = {
			factor = 2.5
			FROM = { trait = ambitious }
		}
		
		modifier = {
			factor = 4.0
			FROM = { 
				OR = {
					trait = lustful 
					trait = hedonist
				}
			}
		}
		
		modifier = {
			factor = 1.5
			FROM = { trait = genius }
		}
		modifier = {
			factor = 1.25
			FROM = { trait = gregarious }
		}
		modifier = {
			factor = 1.25
			FROM = { trait = fair }
		}
		
		modifier = {
			factor = 0.5
			FROM = { trait = honest } 
		}
		
		modifier = {
			factor = 0.5
			FROM = { trait = slow } 
		}
		
		modifier = {
			factor = 0.01
			trait = chaste
		}
		
		modifier = {
			factor = 0.85
			FROM = { age = 30 }
		}
		
		modifier = {
			factor = 0.5
			FROM = { age = 30 }
		}

			FROM = { age = 45 }

			factor = 0.4
			FROM = { num_of_children = 2 }
		}

				FROM = { trait = lunatic }

		}
